def getYouZodiac():
    birth = input("请输入您的出生日期，格式：2020-03-22\n").strip(" ")
    birs = birth.split("-")
    if len(birs) == 3 and str.isnumeric(birs[0]) and str.isnumeric(birs[1]) and str.isnumeric(birs[2]):
        cyear = birs[0]
        cmonth = birs[1]
        cday = birs[2]
    else:
        print("您输入有误请重试！\n")

    for item in ztime:
        if item[:4] == cyear:
            smonth = item.split("-")[1]
            sday = item.split("-")[2]

    if int(cmonth) < int(smonth) or \
            ( int(cmonth) == int(smonth) and
              int(cday) < int(sday) ):
        cyear = int(cyear) - 1

    rem = int(cyear) % 12
    print('要查询的属相是:' + zodiac[rem] + '\n属相对应的年份是:' + lunar[rem] + '年')

if __name__ == '__main__':
    getYouZodiac()
